6 Consultas (Queries)  

# Perfil completo del profesional
query PerfilDeProfesional {
  professional(id: "ID_DEL_PROFESIONAL") {
    firstName
    lastName
    taxId
    services {
      name
    }
    workExperience {
      jobTitle
      company
      startDate
      endDate
      description
    }
    education {
      degree
      institution
      year
    }
  }
}

# Inventario de vacantes
query InventarioDeVacantes {
  vacancies {
    id
    title
    service {
      name
    }
    employer
    location
    createdAt
  }
}

# Resumen de empleador
query ResumenDeEmpleador {
  employerSummary(id: "ID_DEL_EMPLEADOR") {
    name
    taxId
    vacanciesOffered
  }
}

# Postulantes por área (servicio)
query PostulantesPorArea {
  applicantsByService(serviceId: "ID_DEL_SERVICIO") {
    serviceName
    applicants
  }
}

# Estadísticas de profesionales por servicio
query EstadisticasPorServicio {
  professionalsByServiceStats {
    serviceName
    count
    percentage
  }
}

# Estadísticas de profesionales por género
query CantidadPorGenero {
  professionalsByGender {
    gender
    count
  }
}
